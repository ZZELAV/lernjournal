# Parameter: [1] Celsius Eingabe

function celsiusToFahrenheit {
    param (
        # celsius eingabe
        [Parameter(Mandatory = $true)]
        [double]
        $celsius
    )
    
    [double]$fahrenheit = ($celsius * 9) / 5 + 32

    return $fahrenheit
}
```

# Parameter:
# ---

function dateInHashtable {
    # datum holen
    $date = Get-Date

    # hashtable erstellen
    $dateHashtable = @{
        "day" = $date.day
        "month" = $date.month
        "year" = $date.year
    }
    
    return $dateHashtable
}
```
